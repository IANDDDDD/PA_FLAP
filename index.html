<script>

// ====================================================================

// SUPER SIMPLE FLAPPY GAME — guaranteed to work everywhere

// ====================================================================

(function(){

  const c = document.getElementById("game");

  const x = c.getContext("2d");


  const W = c.width;

  const H = c.height;


  // game variables

  let started=false, over=false, score=0, best=0;

  let last=0, spawn=0;


  const bird={x:100,y:H/2,r:10,vy:0};

  const pipes=[];

  

  const G=900;      // gravity

  const FLAP=-280;  // jump force

  const SPEED=120;  // pipe speed

  const GAP=150;    // gap

  const PIPE_W=50;  

  const FLOOR=80;   


  function reset(){

    started=false;

    over=false;

    score=0;

    bird.x=100; bird.y=H/2; bird.vy=0;

    pipes.length=0;

    last=performance.now();

    draw(0);

  }


  function flap(){

    if(over){ reset(); return; }

    started=true;

    bird.vy=FLAP;

  }


  c.addEventListener("pointerdown", flap);

  window.addEventListener("keydown", e=>{

    if(e.code==="Space"||e.code==="ArrowUp"){

      e.preventDefault(); flap();

    }

  });


  function addPipe(){

    const topH = 40 + Math.random()*(H - FLOOR - GAP - 80);

    pipes.push({x:W+PIPE_W, topH, passed:false});

  }


  function update(dt){

    if(!started || over) return;


    bird.vy += G*dt;

    bird.y  += bird.vy*dt;


    spawn+=dt;

    if(spawn>1.2){ spawn=0; addPipe(); }


    for(const p of pipes){

      p.x -= SPEED*dt;

      if(!p.passed && p.x+PIPE_W < bird.x-bird.r){

        p.passed=true;

        score++;

        best=Math.max(best,score);

      }

    }


    while(pipes.length && pipes[0].x < -PIPE_W-10) pipes.shift();


    const ground=H-FLOOR;

    if(bird.y-bird.r<0 || bird.y+bird.r>ground) over=true;


    for(const p of pipes){

      if(bird.x+bird.r>p.x && bird.x-bird.r<p.x+PIPE_W){

        if(bird.y-bird.r<p.topH || bird.y+bird.r>p.topH+GAP){

          over=true;

        }

      }

    }

  }


  function draw(){

    // background

    x.fillStyle="#0a0d13";

    x.fillRect(0,0,W,H);


    // pipes

    x.fillStyle="#61ffa8";

    for(const p of pipes){

      x.fillRect(p.x, 0, PIPE_W, p.topH);

      x.fillRect(p.x, p.topH+GAP, PIPE_W, H);

    }


    // ground

    x.fillStyle="#0c1018";

    x.fillRect(0,H-FLOOR,W,FLOOR);


    // bird

    x.fillStyle="#ffd166";

    x.beginPath();

    x.arc(bird.x,bird.y,bird.r,0,Math.PI*2);

    x.fill();


    // score

    x.fillStyle="#fff";

    x.font="32px system-ui";

    x.fillText(score,16,40);


    // messages

    if(!started && !over){

      x.font="24px system-ui";

      x.textAlign="center";

      x.fillText("Playing > Posting", W/2, H*0.4);

      x.font="16px system-ui";

      x.fillText("Tap / Space to play", W/2, H*0.47);

    }


    if(over){

      x.textAlign="center";

      x.font="26px system-ui";

      x.fillStyle="#fff";

      x.fillText("Game Over", W/2, H*0.38);     

      x.font="16px system-ui";

      x.fillText("Tap / Space to restart", W/2, H*0.45);

    }

  }


  function loop(t){

    const dt = Math.min(0.03, (t-last)/1000);

    last = t;

    update(dt);

    draw();

    requestAnimationFrame(loop);

  }


  reset();

  requestAnimationFrame(loop);

})();
